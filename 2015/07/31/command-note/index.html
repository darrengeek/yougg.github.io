    • 反射访问私有成员

      package main
    
import (
    "fmt"
    "reflect"
    "unsafe"
)
        
type Tab struct {
    A string
    b int
    c *bool
}
        
func (t *Tab) Name(x string) string {
    return x+`{A:"`+t.A+`",b:`+fmt.Sprintf("%d",t.b)+`,c:`+fmt.Sprintf("%t",*t.c)+`}`
}
        
func main() {
    var t = &Tab{c:new(bool)}
    fmt.Printf("%#v\n",t)
    val := reflect.ValueOf(t)
    vv := reflect.Indirect(val)
    vv.FieldByName("A").SetString("XXXXXX")
    // panic: reflect: reflect.Value.SetInt using value obtained using unexported field
    // vv.FieldByName("b").SetInt(222)
    b := (*int)(unsafe.Pointer(vv.FieldByName("b").Addr().Pointer()))
    *b = 33333
    // panic: reflect: reflect.Value.SetBool using value obtained using unexported field
    // vv.FieldByName("c").SetBool(true)
    c := (*bool)(unsafe.Pointer(vv.FieldByName("c").Pointer()))
    *c = true
    res := val.MethodByName("Name").Call([]reflect.Value{reflect.ValueOf("&main.Tab")})
    for _, v := range res {
        fmt.Println(v)
    }
}

// 读取配置文件加载所有参数
func initConfig() {
	cfgfile := strings.TrimSpace(*config)
	if `` != cfgfile {
		f, err := os.Open(cfgfile)
		if nil == err {
			r := bufio.NewReader(f)
			for {
				line, err := r.ReadString('\n')
				if err == io.EOF {
					break
				}
				if err != nil {
					continue
				}
				line = strings.TrimSpace(line)
				if strings.HasPrefix(line, "#") {
					continue
				}
				kv := strings.Split(line, "=")
				if len(kv) == 2 {
					k := strings.TrimSpace(kv[0])
					v := strings.TrimSpace(kv[1])
					configs[k] = v
				}
			}
		}
	}
}

func checkArgsValid() {
	if flag.NFlag() == 0 {
		flag.Usage()
		waitAndExit(1)
	}

	var i int
	if *isList {
		i++
		configs["action"] = "list"
	}
	if *isBackup {
		i++
		configs["action"] = "backup"
	}
	if *isRestore {
		i++
		configs["action"] = "restore"
	}
	if *isPack {
		i++
		configs["action"] = "pack"
	}
	if *isUpload {
		i++
		configs["action"] = "upload"
	}
	if *isDownload {
		i++
		configs["action"] = "download"
	}
	if i > 1 {
		fmt.Println("Please set only one option of [-backup,-restore,-pack,-upload,-download] for program.")
		flag.Usage()
		waitAndExit(1)
	}

	if configs["action"] != "pack" {
		if "" == *config {
			fmt.Println("Must set option [-config] for configure file .")
			flag.Usage()
			waitAndExit(1)
		}

		if `` != *bucket {
			configs["bucket"] = *bucket
		}

		if `` == configs["bucket"] {
			if configs["action"] != "list" {
				fmt.Println("Bucket name can not be empty.")
				waitAndExit(1)
			}
		} else {
			// 桶名规则: 只能以小写字母或数字开始,长度3~63,只能包含小写字母(a-z),数字(0-9),点(.),减号(-)
			if !bucketreg.MatchString(configs["bucket"]) {
				fmt.Printf("Bucket name %s is invalid.\n", configs["bucket"])
				waitAndExit(1)
			}
		}

		if `` != *period {
			configs["period"] = *period
		}

		switch *showpg {
		case "true", "TRUE":
			configs["showprogress"] = "true"
		case "false", "FALSE":
			configs["showprogress"] = "false"
		default:
			if `` != *showpg {
				configs["showprogress"] = "false"
			}
		}

		initObsAuth()
	}

	if configs["action"] == "backup" {
		if *maxnum >= 0 {
			configs["maxnum"] = strconv.Itoa(*maxnum)
		}
	}

	if configs["action"] != "list" {
		if `` != *src {
			configs["source"] = *src
		}

		srcfile := strings.TrimSpace(configs["source"])
		if `` == srcfile {
			fmt.Printf("Source file path [%s] can not be empty.\n", srcfile)
			waitAndExit(1)
		}

		switch configs["action"] {
		case "backup", "upload":
			_, err := os.Stat(srcfile)
			if nil != err {
				fmt.Printf("Source file path [%s] error, check the file exist and permission to access.\n", srcfile)
				waitAndExit(1)
			}

			switch configs["period"] {
			case periodDay, periodWeek, periodMonth, periodYear:
			default:
				fmt.Println("The period must be one of [DAY,WEEK,MONTH,YEAR]")
				flag.Usage()
				waitAndExit(1)
			}
		}
	}

	if `` != *logfile {
		configs["log"] = *logfile
	}

	logpath := strings.TrimSpace(configs["log"])
	if `` == logpath {
		fmt.Printf("Log file path [%s] can not be empty.\n", logpath)
		waitAndExit(1)
	}
	_, err := os.Stat(logpath)
	if nil != err && !os.IsNotExist(err) {
		fmt.Printf("Log file path [%s] error.\n", logpath)
		waitAndExit(1)
	}
}

// 读取OBS服务器地址,AK,SK 初始化OBS客户端
func initObsAuth() {
	AK := strings.TrimSpace(configs["AK"])
	SK := strings.TrimSpace(configs["SK"])
	endpoint := strings.TrimSpace(configs["endpoint"])

	if `` == AK || `` == SK || `` == endpoint {
		fmt.Printf("The OBS server configuration can not be empty.\n"+
			"Please check if %s exist and AK/SK/endpoint were setted.\n", *config)
		waitAndExit(1)
	}

	obs = client.Factory(AK, SK, endpoint, true)
}

// 初始化日志记录
func initLog() {
	logdir := filepath.Dir(configs["log"])
	err := os.MkdirAll(logdir, os.ModePerm)
	if nil != err {
		fmt.Printf("Create log dir [%s] failed.\n", logdir)
		waitAndExit(1)
	}
	logWriter := log4go.NewFileLogWriter(configs["log"], true)
	log.AddFilter("file", log4go.FINEST, logWriter)
}

// 初始化临时目录
func initTmp() {
	err := os.MkdirAll(tmpdir, os.ModePerm)
	if nil != err {
		log.Error("Create temp dir ", tmpdir, " failed.")
		fmt.Printf("Create temp dir [%s] failed.\n", tmpdir)
		waitAndExit(1)
	}
}

func upload() {
	bucket := configs["bucket"]
	src := configs["source"]
	tarball, ok := configs["tarball"]
	if (!ok) || `` == strings.TrimSpace(tarball) {
		tarball = src
	}
	log.Info("Start upload file %s to bucket %s", tarball, bucket)

	f, err := os.Stat(tarball)
	if nil != err {
		log.Error("Upload source file ", tarball, " not exist.", err)
		waitAndExit(1)
	}

	if !checkBucketExist(bucket) {
		log.Warn("Bucket %s is not exist, create it now.", bucket)
		if !createBucket(bucket) {
			log.Error("Create bucket ", bucket, " failed.")
			fmt.Printf("Create bucket %s failed.\n", bucket)
			waitAndExit(1)
		}
	}

	// source: /opt/huawei/service-manager, Base: service-manager
	dir := filepath.Base(src)
	if pkgnamereg.MatchString(dir) {
		sss := pkgtailreg.Split(dir, 2)
		if `` != sss[0] {
			dir = sss[0]
		}
	}

	ball := filepath.Base(tarball)
	var product string
	switch {
	case (strings.Contains(src, "bigdata") && !strings.Contains(src, "billing/bigdata")) || bigdatareg.MatchString(src):
		product = "bigdata"
	case strings.Contains(src, "inforsight"):
		product = "inforsight"
	case strings.Contains(src, "streamsmart"):
		product = "streamsmart"
	case strings.Contains(src, "billing"):
		product = "billing"
	case strings.Contains(src, "log"):
		product = "database"
	case strings.Contains(src, "iam"):
		product = "iam"
	case strings.Contains(src, "cert"):
		product = "certificate"
	default:
		product = "thirdparty"
	}
	object := filepath.Join(product, dir, configs["period"], ball)
	result, isExist := obs.IsObjectExist(bucket, object)
	log.Debug("Check file exist result: %#v\n", result)
	if result.Err == nil {
		// 重复文件确认
		if isExist {
			for i := 1000; i < 10000; i++ {
				object = filepath.Join(product, dir, configs["period"], strings.Join(strings.Split(ball, ".tar."), "_"+strconv.Itoa(i)[1:]+".tar."))
				result, isExist = obs.IsObjectExist(bucket, object)
				if result.Err == nil {
					if !isExist {
						break
					}
				} else {
					fmt.Printf("Check if file %s is exist in OBS server %s get error.\n", tarball, bucket)
					fmt.Printf("Retry %d times...\n", i+1)
					time.Sleep(time.Second)
					if 2 == i {

						waitAndExit(1)
					}
				}
			}
			if isExist {

				waitAndExit(1)
			}
		}
	} else {
		log.Error("Check if file ", tarball, " is exist in OBS server ", bucket, " get error.")
		fmt.Printf("Check if file %s is exist in OBS server %s get error.\n", tarball, bucket)
		waitAndExit(1)
	}

	input := &models.PutObjectInput{
		Bucket:     bucket,
		Object:     object,
		SourceFile: tarball,
	}

	size := f.Size()
	show := (configs["showprogress"] == "true")
	var bar *pb.ProgressBar
	if show {
		bar = pb.New64(size).Start()
	}

	var output *models.PutObjectOutPut

	fmt.Printf("Upload file %s to bucket %s ...\n", tarball, bucket)
	getCount, reschan, outchan := obs.PutObjectWithProgress(input)
	ticker := time.NewTicker(500 * time.Millisecond).C
loop:
	for {
		select {
		case result = <-reschan:
			output = <-outchan
			break loop
		case <-ticker:
			if show {
				c := getCount()
				if c <= size {
					bar.Set64(c)
				}
			}
		case <-time.After(24 * time.Hour):
			log.Error("Upload file timeout after 24 hours.")
			fmt.Println("Upload file timeout after 24 hours.")
			waitAndExit(1)

	}
	if nil == result.Err && result.StatusCode < http.StatusMultipleChoices {
		if show {
			bar.Set64(size)
			bar.Finish()

		cleanOld()

	log.Debug("Upload file output: %#v", output)
	log.Info("Upload file %s to bucket %s finish.", tarball, bucket)
	fmt.Printf("Upload file to bucket %s path: /%s\n", bucket, object)

// 清理过时的备份文件
func cleanOld() {
	log.Info("Start clean old backup files")
	// 移除临时备份目录中的文件
	defer os.RemoveAll(tmpdir)

	num, err := strconv.Atoi(configs["maxnum"])
	if nil != err {
		switch configs["period"] {
		case periodDay:
			num = 7
		case periodWeek:
			num = 4
		case periodMonth:
			num = 12
		default:
			num = 0
		}
	}

	if num > 0 {
		//获取桶内对象
		input := &models.ListObjectsInput{
			Bucket: configs["bucket"],
		}
		result, output := obs.ListObjects(input)
		if nil != result.Err || result.StatusCode >= http.StatusMultipleChoices || output == nil {
			log.Error("List objects in bucket ", configs["bucket"], " failed.")
			fmt.Printf("List objects in bucket %s failed.\n", configs["bucket"])
			return
		}

		files := make([]string, 0, len(output.Contents))
		dir := filepath.Base(configs["source"])
		object := filepath.Join("bigdata", dir, configs["period"], dir)
		for _, c := range output.Contents {
			// 一天内重复备份的文件暂时不处理, 只清理每天定时备份的文件
			if strings.HasPrefix(c.Key, object) {
				files = append(files, c.Key)
			}
		}
		sort.Strings(files)

		if len(files) > num {
			delFiles := files[:len(files)-num]
			fmt.Printf("Delete old backup files: \n%v\n", delFiles)

			objs := make([]models.Object, 0, len(delFiles))
			for _, f := range delFiles {
				objs = append(objs, models.Object{
					Object: f,
				})
			}
			//批量删除对象
			input := &models.DeleteObjectsInput{
				Bucket:  configs["bucket"],
				Objects: objs,
			}
			result, output := obs.DeleteObjects(input)
			if nil != result.Err || result.StatusCode >= http.StatusMultipleChoices || output == nil {
				log.Error("List objects in bucket ", configs["bucket"], " failed.")
				fmt.Printf("List objects in bucket %s failed.\n", configs["bucket"])
				return
			}
			for _, val := range output.Errors {
				fmt.Printf("Delete old file %s failed, error: %s\n", val.Key, val.Code)
			}
		}
	}
}

// RandString 生成指定长度(length)的随机密码pwd
//  密码长度为 4~64位, 超范围返回空密码
//  密码包含以下必选项:
//  小写字母: a-z
//  大写字母: A-Z
//  印度数字: 0-9
//  特殊符号: @-_
func RandString(length int) (pwd string) {
	if length < 4 || length > 64 {
		return
	}
	// 1. 随机分配4种类型字符的数量
	assign := [...]int{1, 1, 1, 1}
	types := len(assign)
	for i := 0; i < (length - 4); i++ {
		assign[rand.Intn(types)]++
	}
	sum := 0
	for _, v := range assign {
		sum += v
	}
	if length != sum {
		return
	}
	// 2. 根据各类型字符的数量,随机填充密码
	chars := [][]byte{
		[]byte(lowercaseLetter),
		[]byte(specialSymbols),
		[]byte(uppercaseLetter),
		[]byte(digital),
	}
	pwds := make([]byte, length)
	index := 0
	for i := range pwds {
		if types > 1 {
			index = rand.Intn(types)
		} else {
			index = 0
		}
		pwds[i] = chars[index][rand.Intn(len(chars[index]))]
		if assign[index]--; assign[index] <= 0 {
			types--
			assign[index], assign[types] = assign[types], assign[index]
			chars[index], chars[types] = chars[types], chars[index]
		}
	}
	// 3. 随机打乱生成的密码序
	for i := range pwds {
		n := rand.Intn(length)
		pwds[i], pwds[n] = pwds[n], pwds[i]
	}
	pwd = string(pwds)
	return
}

type sftpClient struct {
	host, user, password string
	port                 int
	*sftp.Client
}

// Create a new SFTP connection by given parameters
func NewConn(host, user, password string, port int) (client *sftpClient, err error) {
	switch {
	case `` == strings.TrimSpace(host),
		`` == strings.TrimSpace(user),
		`` == strings.TrimSpace(password),
		0 >= port || port > 65535:
		return nil, errors.New("Invalid parameters")
	}

	client = &sftpClient{
		host:     host,
		user:     user,
		password: password,
		port:     port,
	}

	if err = client.connect(); nil != err {
		return nil, err
	}
	return client, nil
}

func (sc *sftpClient) connect() (err error) {
	config := &ssh.ClientConfig{
		User:    sc.user,
		Auth:    []ssh.AuthMethod{ssh.Password(sc.password)},
		Timeout: 30 * time.Second,
	}

	// connet to ssh
	addr := fmt.Sprintf("%s:%d", sc.host, sc.port)
	conn, err := ssh.Dial("tcp", addr, config)
	if err != nil {
		return err
	}

	// create sftp client
	client, err := sftp.NewClient(conn)
	if err != nil {
		return err
	}
	sc.Client = client

	return nil
}

// Upload file to sftp server
func (sc *sftpClient) Put(localFile, remoteFile string) (err error) {
	srcFile, err := os.Open(localFile)
	if err != nil {
		return
	}
	defer srcFile.Close()

	// Make remote directories recursion
	parent := filepath.Dir(remoteFile)
	path := string(filepath.Separator)
	dirs := strings.Split(parent, path)
	for _, dir := range dirs {
		path = filepath.Join(path, dir)
		sc.Mkdir(path)
	}

	dstFile, err := sc.Create(remoteFile)
	if err != nil {
		return
	}
	defer dstFile.Close()

	_, err = io.Copy(dstFile, srcFile)
	return
}

// Download file from sftp server
func (sc *sftpClient) Get(remoteFile, localFile string) (err error) {
	srcFile, err := sc.Open(remoteFile)
	if err != nil {
		return
	}
	defer srcFile.Close()

	dstFile, err := os.Create(localFile)
	if err != nil {
		return
	}
	defer dstFile.Close()

	_, err = io.Copy(dstFile, srcFile)
	return
